using System;

class Program
{
    static void Main()
    {
        int inicio;
        int fim;

        Console.Write("Digite o número inicial: ");
        inicio = int.Parse(Console.ReadLine());

        Console.Write("Digite o número final: ");
        fim = int.Parse(Console.ReadLine());

        while (inicio > fim)
        {
            Console.WriteLine("Erro! O número inicial deve ser menor que o final.");

            Console.Write("Digite o número inicial novamente: ");
            inicio = int.Parse(Console.ReadLine());

            Console.Write("Digite o número final novamente: ");
            fim = int.Parse(Console.ReadLine());
        }

        for (int i = inicio; i <= fim; i++)
        {
            Console.WriteLine(i);
        }
    }
}

using System;

class Program
{
    static void Main()
    {
        int[] numeros = new int[10];
        int maior = 0;

        for (int i = 0; i < 10; i++)
        {
            Console.Write("Digite um número: ");
            numeros[i] = int.Parse(Console.ReadLine());

            if (i == 0 || numeros[i] > maior)
            {
                maior = numeros[i];
            }
        }

        Console.WriteLine("Maior número digitado: " + maior);
    }
}
